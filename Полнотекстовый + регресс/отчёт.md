
## Проверить, что номер телефона соответствует формату +7(XXX)XXX-XX-XX

```sql
WITH phones AS (
  SELECT '+7(999)123-45-67' AS phone_number
  UNION ALL
  SELECT '8(999)123-45-67' AS phone_number
  UNION ALL
  SELECT '+7(999)1234567' AS phone_number
  UNION ALL
  SELECT '+7(123)456-78-90' AS phone_number
)

SELECT
  phone_number,
  phone_number ~ '^\+7\(\d{3}\)\d{3}-\d{2}-\d{2}$' AS is_valid
FROM phones;

```

## Найти все email-адреса в тексте, игнорируя регистр

```sql
WITH documents AS (
  SELECT 1 AS id, 'Мой email: Test.User@example.com. Пишите!' AS content
  UNION ALL
  SELECT 2 AS id, 'Контакты: info@company.org и SUPPORT@DOMAIN.NET.' AS content
  UNION ALL
  SELECT 3 AS id, 'Здесь нет email адресов.' AS content
)

SELECT
  id,
  -- g - глобальный поиск
  regexp_matches(content, '\b[A-Z0-9._%+-]+@[A-Z0-9.-]+\.[A-Za-z]{2,}\b', 'g') AS email
FROM documents;
```

## Извлечь все хештеги #из_произвольного_текста

```sql

WITH posts AS (
  SELECT 'Отличный день! #погода #солнце' AS post_text
  UNION ALL
  SELECT 'Изучаю #PostgreSQL и #SQL. Очень интересно!' AS post_text
)

SELECT
  regexp_matches(post_text, '#\w+', 'g') AS hashtag
FROM posts;

```

## Заменить все даты в формате DD.MM.YYYY на YYYY-MM-DD

```sql
WITH logs AS (
  SELECT 'Событие произошло 25.09.2025, следующее запланировано на 31.12.2025.' AS log_entry
)

SELECT
  log_entry AS original_text,
  regexp_replace(log_entry, '(\d{2})\.(\d{2})\.(\d{4})', '\3-\2-\1', 'g') AS converted_text
FROM logs;
```


## Написать запрос для поиска строки с % с помощью LIKE

```sql
WITH products AS (
  SELECT 'Скидка 15%' AS name
  UNION ALL
  SELECT '100% хлопок' AS name
  UNION ALL
  SELECT 'Обычный товар' AS name
)

SELECT
  name
FROM products
WHERE name LIKE '%\%%' ESCAPE '\';
```
